    private hashCode(str: string): number {
        let hash = 0;
        for (let i = 0; i < str.length; i++) {
            const char = str.charCodeAt(i);
            hash = ((hash << 5) - hash) + char;
            hash = hash & hash; // ËΩ¨Êç¢‰∏∫32‰ΩçÊï¥Êï∞
        }
        return Math.abs(hash);
    }

    private generateColor(name: string): string {
        const hash = this.hashCode(name);
        // ‰∏∫‰∫ÜÊõ¥Â•ΩÁöÑËßÜËßâÊïàÊûúÔºåÊàë‰ª¨Ë∞ÉÊï¥ÂèÇÊï∞ËåÉÂõ¥
        const hue = hash % 360;
        const saturation = 65 + (hash % 25); // 65-90%
        const lightness = 45 + (hash % 15);  // 45-60%
        
        return `hsl(${hue}, ${saturation}%, ${lightness}%)`;
    }

    public getCategoryColor(categoryName: string): string {

        const index = Object.keys(this.colorMap).length;

        if (index < this.colors.length) {
            color = this.colors[index];
        } else {
            // Âê¶ÂàôÂä®ÊÄÅÁîüÊàêÈ¢úËâ≤
            color = this.generateColor(categoryName);
        }
        
        this.colorMap[categoryName] = color;
        return color;
    }

            const categoryName = link.textContent?.trim();

                const backgroundColor = this.getCategoryColor(categoryName);

                link.style.setProperty('background-color', backgroundColor, 'important');
                link.style.setProperty('color', this.textColor, 'important');
            }
        });
    }
