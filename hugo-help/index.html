        // å¦‚æœå·²ç»æœ‰è¿™ä¸ªåˆ†ç±»çš„é¢œè‰²ï¼Œç›´æ¥è¿”å›
        if (this.colorMap[categoryName]) {
            return this.colorMap[categoryName];
        }

        // è·å–åˆ†ç±»ç´¢å¼•ï¼ˆåŸºäºå·²åˆ†é…é¢œè‰²çš„æ•°é‡ï¼‰
        const index = Object.keys(this.colorMap).length;
        
        let color: string;
        // å¦‚æœç´¢å¼•åœ¨é¢„å®šä¹‰é¢œè‰²èŒƒå›´å†…ï¼Œä½¿ç”¨é¢„å®šä¹‰é¢œè‰²
        if (index < this.colors.length) {
            color = this.colors[index];
        } else {
            // å¦åˆ™åŠ¨æ€ç”Ÿæˆé¢œè‰²
            color = this.generateColor(categoryName);
        }
        
        this.colorMap[categoryName] = color;
        return color;
    }

    // åº”ç”¨é¢œè‰²åˆ°æ‰€æœ‰åˆ†ç±»æ ‡ç­¾
    public applyCategoryColors(): void {
        // æŸ¥æ‰¾æ‰€æœ‰åˆ†ç±»æ ‡ç­¾
        // const categoryElements = document.querySelectorAll('.article-category a, .tagCloud-tags a, .widget-categories a');
        const categoryElements = document.querySelectorAll('.article-category a');
        
        // éå†æ‰€æœ‰åˆ†ç±»æ ‡ç­¾
        categoryElements.forEach(element => {
            const link = element as HTMLAnchorElement;
            // è·å–åˆ†ç±»åç§°ï¼ˆé“¾æ¥æ–‡æœ¬ï¼‰
            const categoryName = link.textContent?.trim();
            
            if (categoryName) {
                // è·å–å¹¶åº”ç”¨é¢œè‰²
                const backgroundColor = this.getCategoryColor(categoryName);
                
                // åº”ç”¨æ ·å¼
                link.style.backgroundColor = backgroundColor;
                link.style.color = this.textColor;
                
                // ç¡®ä¿æ ·å¼ä¼˜å…ˆçº§
                link.style.setProperty('background-color', backgroundColor, 'important');
                link.style.setProperty('color', this.textColor, 'important');
            }
        });
    }
}
